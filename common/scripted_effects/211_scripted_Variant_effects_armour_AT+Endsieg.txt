	make_Puma =
	{
		if =
		{
			limit =
			{
				NOT = { has_country_flag = has_Puma}
			}
			set_country_flag = has_Puma

			{
				limit =
				{

					set_technology = 
					{	
						light_td4	= 1
						light_art4	= 1
						light_spaa4	= 1
						popup = no
					}
				}
			}
			create_equipment_variant = 
			{
				name = "Puma"
				type = light_tank_equipment_4
				upgrades = 
				{
					tank_gun_upgrade = 20
					tank_armor_upgrade = 15
					tank_engine_upgrade = 1000
					tank_reliability_upgrade = 20
					ZMM_LAND_prod_upgrade = var:Pz_discount
				}
			}
			create_equipment_variant = 
			{
				name = "LSFL IV"
				type = light_tank_artillery_equipment_4
				upgrades = 
				{
					tank_art_upgrade = 20
					tank_armor_upgrade = 15
					tank_engine_upgrade = 1000
					tank_reliability_upgrade = 20
					ZMM_LAND_prod_upgrade = var:Pz_discount
				}
			}
			create_equipment_variant = 
			{
				name = "E-10A5"
				type = light_tank_destroyer_equipment_4
				upgrades = 
				{
					tank_at_upgrade = 20
					tank_armor_upgrade = 15
					tank_engine_upgrade = 1000
					tank_reliability_upgrade = 20
					ZMM_LAND_prod_upgrade = var:Pz_discount
				}
			}
			create_equipment_variant = 
			{
				name = "Gepard 1A5"
				type = light_tank_aa_equipment_4
				upgrades = 
				{
					tank_aa_upgrade = 20
					tank_armor_upgrade = 15
					tank_engine_upgrade = 1000
					tank_reliability_upgrade = 20
					ZMM_LAND_prod_upgrade = var:Pz_discount
				}
			}
		}
		every_country =
		{
			limit = 
			{
				is_puppet_of = ROOT
				NOT = { has_country_flag = has_Puma}
				NOT = 
				{
					has_license = #Tank
					{
						FROM = OVERLORD
						equipment = 
						{
							type = light_tank_equipment_4
							version = 1
						}
					}
					has_license = #SPG
					{
						FROM = OVERLORD
						equipment = 
						{
							type = light_tank_artillery_equipment_4
							version = 1
						}
					}
					has_license = #TD
					{
						FROM = OVERLORD
						equipment = 
						{
							type = light_tank_destroyer_equipment_4
							version = 1
						}
					}
					has_license = #SPAA
					{
						FROM = OVERLORD
						equipment = 
						{
							type = light_tank_aa_equipment_4
							version = 1
						}
					}
				}
			}
			ROOT =
			{
				create_production_license =	#Tank
				{
					target = PREV
					equipment = 
					{
						type	= light_tank_equipment_4
						version = 1
					}
					cost_factor = 0
				}
				create_production_license =	#SPG
				{
					target = PREV
					equipment = 
					{
						type	= light_tank_artillery_equipment_4
						version = 1
					}
					cost_factor = 0
				}
				create_production_license =	#TD
				{
					target = PREV
					equipment = 
					{
						type	= light_tank_destroyer_equipment_4
						version = 1
					}
					cost_factor = 0
				}
				create_production_license =	#SPAA
				{
					target = PREV
					equipment = 
					{
						type	= light_tank_aa_equipment_4
						version = 1
					}
					cost_factor = 0
				}
			}
		}
	}
