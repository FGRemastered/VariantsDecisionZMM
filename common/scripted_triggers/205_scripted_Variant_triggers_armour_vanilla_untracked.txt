	can_VWF =
	{
		has_tech = motorised_infantry 
		has_variable = Pz_discount
		any_country =
		{
			OR =
			{
				tag = ROOT
				is_puppet_of = ROOT
			}
			NOT = 
			{
				OR =
				{
					has_country_flag = has_VWF
					has_license =
					{
						from = OVERLORD
						equipment = 
						{
							type = motorized_equipment_1
							version = 1
						}
					}
				}
			}
		}
	}
